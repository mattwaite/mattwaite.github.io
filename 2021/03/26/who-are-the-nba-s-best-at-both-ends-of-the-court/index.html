good <- cleanplayers %>% filter(DRtg < 106 & ORtg > 125)
ggplot() + geom_point(data=cleanplayers, aes(x=ORtg, y=DRtg), color="grey") + geom_point(data=good, aes(x=ORtg, y=DRtg), color="red") + geom_text_repel(data=good, aes(x=ORtg, y=DRtg, label=CleanName)) + theme_minimal() + labs(title="The best two-way players by rating are centers", subtitle="And one of them plays for the Knicks.
